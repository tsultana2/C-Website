Car::Car(int ml,int st,const char* clr,const *char nm)
{
bool valid = true;
if(valid){
miles = ml;
seat = st;
color = new char[strlen(clr)+1];
strcpy(color,clr);
name = new char[strlen(nm)+1];
strcpy(name,nm);
}
else
miles = 0;
seat = 0;
color = nullptr;
name = nullptr;
}
//Copy
Car::Car(const Car& src);
{
miles = src.miles;
seat = src.seat;
if(src.color!=nullptr){
color = new char[strlen(clr)+1];
strcpy(color,clr);
}
else
color = nullptr;
if(src.name!=nullptr){
name = new char[strlen(src.name)+1];
strcpy(name,src.name);
}
else
name = nullptr;
}
//Assignment
Car& Car::operator(const Car& src);
{
if(this!=&src){
miles = src.miles;
seat = src.seat;
delete [] color; if(src.color!=nullptr){
color = new char[strlen(clr)+1];
strcpy(color,clr);
}
else
color = nullptr;
delete [] name;
if(src.name!=nullptr){
name = new char[strlen(src.name)+1];
strcpy(name,src.name);
}
else
name = nullptr;
}
return *this;
}
