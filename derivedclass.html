Inheritance



Inheritence is the rlationship between classes where
one class inherits the entire structure of another class.
This relationship is more stronger than composition.

// Student.h
#include &lt;ostream&gt;
const int N = 30;
const int M = 13;
// Base Class - start
class Person {
char person[N+1];
public:
void set(const char* n);
void displayName(std::ostream&) const;
};// Base Class - end
 // Derived Class - start
class Student : public Person {
int no;
char grade[M+1];
public:
Student();
Student(int, const char*);
void display(std::ostream&) const;
};// Derived Class - end
